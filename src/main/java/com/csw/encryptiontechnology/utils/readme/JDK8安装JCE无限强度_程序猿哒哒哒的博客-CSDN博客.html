  1. package cn.exrick.xboot.core.common.utils;
  2.  
  3. import lombok.extern.slf4j.Slf4j;
  4. import org.jasypt.encryption.pbe.PooledPBEStringEncryptor;

  6.  
  7.

  9.  * @author Exrickx
  10.  */
  11. @Slf4j
  12. public class JasyptUtil {
  13.  
  14.     /**
  15.      * Jasypt生成加密结果
  16.      * @param password 配置文件中设定的加密密码 jasypt.encryptor.password
  17.      * @param value 待加密值
  18.      * @return
  19.      */
  20.     public static String encyptPwd(String password,String value){
  21.         PooledPBEStringEncryptor encryptor = new PooledPBEStringEncryptor();
  22.         encryptor.setConfig(cryptor(password));
  23.         String result = encryptor.encrypt(value);
  24.         return result;
  25.     }
  26.  
  27.     /**
  28.      * 解密
  29.      * @param password 配置文件中设定的加密密码 jasypt.encryptor.password
  30.      * @param value 待解密密文
  31.      * @return
  32.      */
  33.     public static String decyptPwd(String password,String value){
  34.         PooledPBEStringEncryptor encryptor = new PooledPBEStringEncryptor();
  35.         encryptor.setConfig(cryptor(password));
  36.         encryptor.decrypt(value);
  37.         String result = encryptor.decrypt(value);
  38.         return result;
  39.     }
  40.  
  41.     public static SimpleStringPBEConfig cryptor(String password){
  42.         SimpleStringPBEConfig config = new SimpleStringPBEConfig();
  43.         config.setPassword(password);
  44.         config.setAlgorithm("PBEWITHHMACSHA512ANDAES_256");
  45.         config.setKeyObtentionIterations("1000");
  46.         config.setPoolSize(1);
  47.         config.setProviderName("SunJCE");
  48.         config.setSaltGeneratorClassName("org.jasypt.salt.RandomSaltGenerator");
  49.         config.setIvGeneratorClassName("org.jasypt.iv.RandomIvGenerator");
  50.         config.setStringOutputType("base64");
  51.         return config;
  52.     }
  53.  
  54.     public static void main(String[] args){
  55.  
  56.         //加密 若修改了第一个参数加密password记得在配置文件同步修改
  57.         System.out.println(encyptPwd("xboot","123456"));
  58.         //解密
  59.         System.out.println(decyptPwd("xboot",
  60. "ZUNmpH3qrSfC8MGvcT0dEBuJvTl9v9XOMc2cmNj1KH8sQfPmr081eJJlS3ksrpn8"));
  61.     }
  62. }
